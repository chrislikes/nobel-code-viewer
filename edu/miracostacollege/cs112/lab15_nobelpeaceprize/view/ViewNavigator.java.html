3    import javafx.scene.Scene;
4    import javafx.stage.Stage;
5    
6    /**
7     * This static class is used to navigate between scenes (e.g. MainScene, AddScene and/or ModifyScene).
8     *
9     * @author Michael Paulding
10    * @version 1.0
11    */
12   public class ViewNavigator {
13   
14       private static Stage mainStage;
15   
16       /**
17        * Sets the main stage (window) when the application first loads (in the View class)
18        *
19        * @param stage The main stage
20        */
21       public static void setStage(Stage stage) {
22           mainStage = stage;
23       }
24   
25       /**
26        * Navigates from one scene to another, e.g. from the MainScene to the AddScene.
27        *
28        * @param title The title to display on the stage
29        * @param scene The new scene to load
30        */
31       public static void loadScene(String title, Scene scene) {
32           mainStage.setTitle(title);
33           mainStage.setScene(scene);
34           mainStage.show();
35       }
36   
37   }
