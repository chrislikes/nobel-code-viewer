4    import edu.miracostacollege.cs112.lab15_nobelpeaceprize.controller.Controller;
5    import javafx.application.Application;

9    /**
10    * The <code>View</code> class represents the user interface for the application.  It initializes the
11    * primary stage (window), then instructs the ViewNavigator to load the very first scene (the MainScene).
12    *
13    * @author Michael Paulding
14    * @version 1.2
15    */
16   public class View extends Application {
17   
18       /**
19        * Starts the application by setting the stage (window) with a custom Nobel icon, then navigating
20        * to the first scene (canvas), which happens to be the MainScene for this application.
21        *
22        * @param primaryStage The primary stage (window)
23        * @throws Exception if one occurs during startup.
24        */
25       @Override
26       public void start(Stage primaryStage) throws Exception {
27           ViewNavigator.setStage(primaryStage);
28           //TODO: Uncomment when res folder has been configured:
29           primaryStage.getIcons().add(new Image("nobel_peace_prize_icon.png"));
30           ViewNavigator.loadScene("Nobel Peace Prize Laureates", new MainScene());
31       }
32   
33       /**
34        * Stop is called whenever the application shuts down (e.g. user closes the window).
35        * In this case we request the Controller to initiate the saving of all laureate data to the binary file.
36        *
37        * @throws Exception if one occurs during shutdown.
38        */
39       @Override
40       public void stop() throws Exception {
41           Controller.getInstance().saveData();
42       }
43   
44       /**
45        * The entry point to this JavaFX application.  Application.launch will eventually make a call
46        * to the start() method.
47        *
48        * @param args Command line arguments
49        */
50       public static void main(String[] args) {
51           Application.launch(args);
52   
53       }
54   
55   }
