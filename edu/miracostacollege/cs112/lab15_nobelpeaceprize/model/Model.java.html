3    import javafx.collections.FXCollections;
4    import javafx.collections.ObservableList;
5    
6    import java.io.*;
7    import java.nio.file.FileSystems;
8    import java.util.Arrays;
9    import java.util.Scanner;

11   /**
12    * The <code>Model</code> class represents the business logic (data and calculations) of the application.
13    * In the Nobel Peace Prize Laureates app, it either loads laureates from a CSV file (first load) or a binary file (all
14    * subsequent loads).  It is also responsible for saving data to a binary file.
15    *
16    * @author Michael Paulding
17    * @version 1.0
18    */
19   public class Model {
20   
21       public static final String CSV_FILE = "NobelPeacePrizeWinners.csv";
22       public static final String BINARY_FILE = "Laureates.dat";
23   
24   
25       /**
26        * Determines whether the binary file exists and has data (size/length > 5L bytes).
27        *
28        * @return True if the binary file exists and has data, false otherwise.
29        */
30       public static boolean binaryFileHasData() {
31           File binaryFile = new File(BINARY_FILE);
32           return (binaryFile.exists() && binaryFile.length() >= 5);
33       }
34   
35       /**
36        * Populates the list of all laureates from the binary file. This will only be called once, the first time the app
37        * loaded to seed initial data from the CSV file.  All subsequent loads will be extracted from
38        * the binary file.be called everytime the application loads,
39        *
40        * @return The list of all laureates populated from the CSV file
41        */
42       public static ObservableList<NobelLaureate> populateListFromCSVFile() {
43           ObservableList<NobelLaureate> allLaureatesList = FXCollections.observableArrayList();
44           //open csv file and loop through it and add the nobelwinners to list
45           try {
46               Scanner csvFile = new Scanner(new File(CSV_FILE));
47               //skipline
48               csvFile.nextLine();
49               String line, name, country, motivation;
50               int year;
51               double prizeAmount;
52               String[] parts;
53               while (csvFile.hasNextLine()) {
54                   line = csvFile.nextLine();
55                   parts = line.split(",");
56                   name = parts[13];
57                   country = parts[26];
58                   motivation = parts[9];
59                   year = Integer.parseInt(parts[0]);
60                   prizeAmount = Double.parseDouble(parts[6]);
61   
62                   allLaureatesList.add(new NobelLaureate(name, year, motivation, country, prizeAmount));
63               }
64               csvFile.close();
65           } catch (FileNotFoundException e) {
66               System.err.println(e.getMessage());
67           }
68   
69           return allLaureatesList;
70       }
71   
72       /**
73        * Populates the list of all laureates from the binary file. This will be called everytime the application loads,
74        * other than the very first time, since it needs initial data from CSV.
75        *
76        * @return The list of all laureates populated from the binary file
77        */
78       public static ObservableList<NobelLaureate> populateListFromBinaryFile() {
79           ObservableList<NobelLaureate> allLaureatesList = FXCollections.observableArrayList();
80           try {
81               ObjectInputStream fileReader = new ObjectInputStream(new FileInputStream(BINARY_FILE));
82               //read file to array
83               NobelLaureate[] laureatesArray = (NobelLaureate[]) fileReader.readObject();
84               allLaureatesList.addAll(Arrays.asList(laureatesArray));
85   
86               fileReader.close();
87   
88           } catch (IOException | ClassNotFoundException e) {
89               System.err.println(e.getMessage());
90           }
91           return allLaureatesList;
92       }
93   
94       /**
95        * Saves the list of all laureates to the binary file. This will be called each time the application stops,
96        * which occurs when the user exits/closes the app.  Note this method is called in the View, by the controller,
97        * during the stop() method.
98        *
99        * @return True if the data were saved to the binary file successfully, false otherwise.
100       */
101      public static boolean writeDataToBinaryFile(ObservableList<NobelLaureate> allLaureatesList) {
102          try {
103              ObjectOutputStream fileWriter = new ObjectOutputStream(new FileOutputStream(BINARY_FILE));
104  
105              NobelLaureate[] laureatesArray = new NobelLaureate[allLaureatesList.size()];
106              allLaureatesList.toArray(laureatesArray);
107              fileWriter.writeObject(laureatesArray);
108              fileWriter.close();
109  
110          } catch (IOException e) {
111              System.err.println(e.getMessage());
112              return false;
113          }
114          return true;
115      }
