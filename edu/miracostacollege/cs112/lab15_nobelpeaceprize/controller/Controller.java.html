9    /**
10    * The <code>Controller</code> is a Singleton object that relays all commands between the Model and View
11    * (and vice versa).  There is only one Controller object, accessible by a call to the static getInstance()
12    * method.
13    *
14    * @author Michael Paulding
15    * @version 1.0
16    */
17   public class Controller {
18   
19       private static Controller theInstance;
20   
21       private Controller() {
22       }
23   
24       //make public method called getInstance
25       public static Controller getInstance() {
26           if (theInstance == null) {
27               theInstance = new Controller();
28               if (Model.binaryFileHasData())
29                   theInstance.allLaureatesList = Model.populateListFromBinaryFile();
30   
31               else
32                   theInstance.allLaureatesList = Model.populateListFromCSVFile();
33           }
34           return theInstance;
35       }
36   
37       private ObservableList<NobelLaureate> allLaureatesList;
38   
39       /**
40        * Gets the list of all laureates.
41        *
42        * @return The list of all laureates.
43        */
44       public ObservableList<NobelLaureate> getAllLaureates() {
45           return allLaureatesList;
46       }
47   
48       /**
49        * Makes a request for the model to save all the laureates data (the list of all laureates) to
50        * a persistent binary file.
51        */
52       public void saveData() {
53   
54           Model.writeDataToBinaryFile(allLaureatesList);
55       }
56   }
